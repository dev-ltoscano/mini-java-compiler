cmake_minimum_required (VERSION 3.5)

set (CMAKE_CXX_STANDARD 11)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR}/build)

set(CMAKE_BUILD_TYPE "Release")

file(GLOB_RECURSE HELPER_H_FILES "${PROJECT_SOURCE_DIR}/helper/*.h")
file(GLOB_RECURSE HELPER_CPP_FILES "${PROJECT_SOURCE_DIR}/helper/*.cpp")

file(GLOB COMPILER_H_FILES "${PROJECT_SOURCE_DIR}/*/*.h")
file(GLOB COMPILER_CPP_FILES "${PROJECT_SOURCE_DIR}/*/*.cpp")

add_executable(mjavac "MiniJavaCompiler.cpp" ${COMPILER_H_FILES} ${COMPILER_CPP_FILES} ${HELPER_H_FILES} ${HELPER_CPP_FILES})
